### 三数之和

```js
var threeSum = function (nums) {
  nums.sort((a, b) => a - b);
  const ans = [];
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] > 0) return ans;
    if (nums[i] === nums[i - 1]) continue;
    let l = i + 1,
      r = nums.length - 1;
    while (l < r) {
      let lNum = nums[l],
        rNum = nums[r],
        iNum = nums[i];
      let threeSum = lNum + rNum + iNum;
      if (threeSum < 0) {
        l++;
      } else if (threeSum > 0) {
        r--;
      } else {
        ans.push([iNum, lNum, rNum]);
        while (nums[l] === nums[l + 1]) l++;
        while (nums[r] === nums[r - 1]) r--;
        l++;
        r--;
      }
    }
  }
  return ans;
};
```
