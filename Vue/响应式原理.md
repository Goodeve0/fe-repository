### Vue2 响应式原理

```js
/**
 * 这里的函数 defineReactive 用来对 Object.defineProperty 进行封装。
 **/
function defineReactive(data, key, val) {
  // 依赖存储的地方
  const dep = new Dep();
  Object.defineProperty(data, key, {
    enumerable: true,
    configurable: true,
    get: function () {
      // 在 getter 中收集依赖
      dep.depend();
      return val;
    },
    set: function (newVal) {
      val = newVal;
      // 在 setter 中触发依赖
      dep.notify();
    },
  });
}
```
